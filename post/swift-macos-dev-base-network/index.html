swift macos 应用开发基础之网络请求

最近在开发flomo-mac，使用到了Alamofire，记录下,本文不对Alamofire进行详细介绍，如果需要可以前往Alamofire Gihub

封装 Alamofire


import Alamofire
import Foundation

class Api {
    let url = "https://flomoapp.com/api/memo/"

    let headers: HTTPHeaders = [
        "X-XSRF-TOKEN": ViewModel().token,
    ]

    func putMemo(parameters: [String : Any], success: @escaping (_ response: Data) -> Void, failure: @escaping (_ error: Error) -> Void) {
        AF.request(url,  method: .put, parameters: parameters, encoding: JSONEncoding.default, headers: headers).validate().responseJSON { response in
            switch response.result {
            case .success:
                if let data = response.data {
                    success(data)
                }
            case let .failure(error):
                failure(error)
            }
        }
    }
}
